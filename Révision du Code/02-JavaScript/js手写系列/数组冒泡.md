### 方法一：从1开始

```js
// 方法一：从1开始
var arr = [4, 3, 2, 7, 5, 1]
for (var i = 1; i < arr.length; i++) {
  for (var j = 0; j < arr.length; j++) {
    if (arr[j] > arr[j + 1]) {
      var tmp = arr[j]
      arr[j] = arr[j + 1]
      arr[j + 1] = tmp
    }
  }
}
console.log(arr)
```
